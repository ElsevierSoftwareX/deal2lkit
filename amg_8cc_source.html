161 template<int dim, int spacedim, typename Matrix>
162 void ParsedAMGPreconditioner::initialize_preconditioner( const Matrix &matrix,
163  const ParsedFiniteElement<dim, spacedim> &fe,
164  const DoFHandler<dim, spacedim> &dh)
165 {
166  TrilinosWrappers::PreconditionAMG::AdditionalData data;
167 
168  data.elliptic = elliptic;
169  data.higher_order_elements = higher_order_elements;
170  data.n_cycles = n_cycles;
171  data.w_cycle = w_cycle;
172  data.aggregation_threshold = aggregation_threshold;
173  if (var_const_modes == "none")
174  {
175  data.constant_modes = std::vector<std::vector<bool> > (0);
176  }
177  else
178  {
179  std::vector< std::vector< bool > > constant_modes;
180  unsigned int pos = fe.get_first_occurence(var_const_modes);
181  bool is_vec = fe.is_vector(var_const_modes);
182  if (is_vec)
183  {
184  FEValuesExtractors::Vector components(pos);
185  DoFTools::extract_constant_modes<DoFHandler<dim, spacedim> > (dh, dh.get_fe().component_mask(components),
186  constant_modes);
187  }
188  else
189  {
190  FEValuesExtractors::Scalar components(pos);
191  DoFTools::extract_constant_modes<DoFHandler<dim, spacedim> > (dh, dh.get_fe().component_mask(components),
192  constant_modes);
193  }
194  data.constant_modes = constant_modes;
195  }
196  data.smoother_sweeps = smoother_sweeps;
197  data.smoother_overlap = smoother_overlap;
198  data.output_details = output_details;
199  data.smoother_type = smoother_type.c_str();
200  data.coarse_type = coarse_type.c_str();
201  this->initialize(matrix, data);
202 }
