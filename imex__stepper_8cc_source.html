132 template <typename VEC>
133 unsigned int IMEXStepper<VEC>::start_ode(VEC &solution, VEC &solution_dot)
134 {
135  AssertDimension(solution.size(), interface.n_dofs());
136 
137  unsigned int step_number = 0;
138 
139 
140  auto previous_solution = interface.create_new_vector();
141  auto solution_update = interface.create_new_vector();
142  auto residual = interface.create_new_vector();
143  auto rhs = interface.create_new_vector();
144 
145  *previous_solution = solution;
146 
147  double t = initial_time;
148  double alpha;
149 
150  // check if it is a stationary problem
151  if (initial_time == final_time)
152  alpha = 0.0;
153  else
154  alpha = 1./step_size;
155 
156  interface.output_step( 0, solution, solution_dot, 0, step_size);
157 
158  // Initialization of the state of the boolean variable
159  // responsible to keep track of the requirement that the
160  // system's Jacobian be updated.
161  bool update_Jacobian = true;
162 
163  bool restart=false;
164  // The overall cycle over time begins here.
165  for (; t<=final_time+1e-15; t+= step_size, ++step_number)
166  {
167  pout << "Time = " << t << std::endl;
168  // Implicit Euler scheme.
169  solution_dot = solution;
170  solution_dot -= *previous_solution;
171  solution_dot *= alpha;
172 
173  // Initialization of two counters for the monitoring of
174  // progress of the nonlinear solver.
175  unsigned int inner_iter = 0;
176  unsigned int outer_iter = 0;
177  unsigned int nonlin_iter = 0;
178  interface.residual(t, solution, solution_dot, *residual);
179  double res_norm = 0.0;
180  double solution_norm = 0.0;
181 
182  if (abs_tol>0.0||rel_tol>0.0)
183  res_norm = interface.vector_norm(*residual);
184  // if (rel_tol>0.0)
185  // solution_norm = interface.vector_norm(solution);
186 
187  // The nonlinear solver iteration cycle begins here.
188  while (outer_iter < max_outer_non_linear_iterations &&
189  res_norm > abs_tol &&
190  res_norm > rel_tol*solution_norm)
191  {
192  outer_iter += 1;
193  if (update_Jacobian == true)
194  {
195  interface.setup_jacobian(t, solution, solution_dot,
196  *residual, alpha);
197  }
198 
199  inner_iter = 0;
200  while (inner_iter < max_inner_non_linear_iterations &&
201  res_norm > abs_tol &&
202  res_norm > rel_tol*solution_norm)
203  {
204 
205 
206  inner_iter += 1;
207 
208  *rhs = *residual;
209  *rhs *= -1.0;
210 
211  interface.solve_jacobian_system(t, solution, solution_dot,
212  *residual, alpha,
213  *rhs, *solution_update);
214 
215 
216  if (method == "LS_backtracking")
217  {
218  newton_alpha = line_search_with_backtracking(*solution_update,
219  *previous_solution,
220  alpha,
221  t,
222  solution,
223  solution_dot,
224  *residual);
225  }
226  else if (method == "fixed_alpha")
227  {
228  solution.sadd(1.0,
229  newton_alpha, *solution_update);
230 
231  // Implicit Euler scheme.
232  solution_dot = solution;
233  solution_dot -= *previous_solution;
234  solution_dot *= alpha;
235  }
236 
237  res_norm = interface.vector_norm(*solution_update);
238 
239  if (rel_tol>0.0)
240  {
241  solution_norm = interface.vector_norm(solution);
242 
243  if (verbose)
244  {
245  pout << std::endl
246  << " "
247  << " iteration "
248  << nonlin_iter + inner_iter
249  << ":\n"
250  << std::setw(19) << std::scientific << res_norm
251  << " update norm\n"
252  << std::setw(19) << std::scientific << solution_norm
253  << " solution norm\n"
254  << std::setw(19) << newton_alpha
255  << " newton alpha\n\n"
256  << std::endl;
257  }
258  }
259  else if (verbose)
260  {
261  pout << std::endl
262  << " "
263  << " iteration "
264  << nonlin_iter + inner_iter
265  << ":\n"
266  << std::setw(19) << std::scientific << res_norm
267  << " update norm\n"
268  << std::setw(19) << newton_alpha
269  << " newton alpha\n\n"
270  << std::endl;
271  }
272 
273  interface.residual(t,solution,solution_dot,*residual);
274  }
275 
276  nonlin_iter += inner_iter;
277 
278  if (std::fabs(res_norm) < abs_tol ||
279  std::fabs(res_norm) < rel_tol*solution_norm)
280  {
281  pout << std::endl
282  << " "
283  << std::setw(19) << std::scientific << res_norm
284  << " (converged in "
285  << nonlin_iter
286  << " iterations)\n\n"
287  << std::endl;
288  break; // Break of the while cycle ... after this a time advancement happens.
289  }
290  else if (outer_iter == max_outer_non_linear_iterations)
291  {
292  pout << std::endl
293  << " "
294  << std::setw(19) << std::scientific << res_norm
295  << " (not converged in "
296  << std::setw(3) << nonlin_iter
297  << " iterations)\n\n"
298  << std::endl;
299  AssertThrow(false,
300  ExcMessage ("No convergence in nonlinear solver"));
301  }
302 
303  } // The nonlinear solver iteration cycle ends here.
304 
305  restart = interface.solver_should_restart(t,step_number,step_size,solution,solution_dot);
306 
307  if (restart)
308  {
309  previous_solution = interface.create_new_vector();
310  solution_update = interface.create_new_vector();
311  residual = interface.create_new_vector();
312  rhs = interface.create_new_vector();
313 
314  t -= step_size;
315  --step_number;
316  }
317  else
318  {
319 
320  if ((step_number % output_period) == 0)
321  interface.output_step(t, solution, solution_dot, step_number, step_size);
322  }
323 
324  *previous_solution = solution;
325  update_Jacobian = update_jacobian_continuously;
326 
327  } // End of the cycle over time.
328  return 0;
329 }

333 template <typename VEC>
334 double IMEXStepper<VEC>::
335 line_search_with_backtracking(const VEC &update,
336  const VEC &previous_solution,
337  const double &alpha,
338  const double &t,
339  VEC &solution,
340  VEC &solution_dot,
341  VEC &residual)
342 {
343  auto first_trial = interface.create_new_vector();
344  auto first_residual = interface.create_new_vector();
345 
346  *first_trial = solution;
347  double n_alpha = 1.0;
348 
349  first_trial->sadd(1.0, n_alpha, update);
350 
351  solution_dot = *first_trial;
352  solution_dot -= previous_solution;
353  solution_dot *= alpha;
354 
355  interface.residual(t, *first_trial, solution_dot, *first_residual);
356 
357  double first_res_norm = interface.vector_norm(*first_residual);
358 
359  auto second_trial = interface.create_new_vector();
360  auto second_residual = interface.create_new_vector();
361 
362  for (unsigned int i=1; i<=n_max_backtracking; ++i)
363  {
364  *second_trial = solution;
365 
366  n_alpha = 1.0/(std::pow(2.0,i));
367 
368  second_trial->sadd(1.0, n_alpha, update);
369 
370  solution_dot = *second_trial;
371  solution_dot -= previous_solution;
372  solution_dot *= alpha;
373 
374  interface.residual(t, *second_trial, solution_dot, *second_residual);
375  double second_res_norm = interface.vector_norm(*second_residual);
376  if (first_res_norm < second_res_norm)
377  {
378  solution = *first_trial;
379  residual = *first_residual;
380 
381  solution_dot = *first_trial;
382  solution_dot -= previous_solution;
383  solution_dot *= alpha;
384  n_alpha = 1.0/(std::pow(2.0,i-1));
385  return n_alpha;
386 
387  }
388  else if (i<(n_max_backtracking))
389  {
390  *first_trial = *second_trial;
391  *first_residual = *second_residual;
392  first_res_norm = second_res_norm;
393  }
394  else
395  {
396  solution = *second_trial;
397  residual = *second_residual;
398  /* solution dot is ok */
399  return n_alpha;
400  }
401  }
402  return n_alpha;
403 }
