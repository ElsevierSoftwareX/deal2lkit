138 template <int dim, int spacedim>
139 template<typename VECTOR>
140 void ParsedDataOut<dim,spacedim>::add_data_vector(const VECTOR &data_vector,
141  const std::string &desc)
142 {
143  AssertThrow(initialized, ExcNotInitialized());
144  deallog.push("AddingData");
145  std::vector<std::string> dd = Utilities::split_string_list(desc);
146  if (data_out->default_suffix() != "")
147  {
148  if (dd.size() ==1 )
149  {
150  data_out->add_data_vector (data_vector, desc);
151  }
152  else
153  {
154  std::vector<std::string>::iterator sit = dd.begin();
155  std::vector<int> occurrances;
156 
157  for (; sit!=dd.end(); ++sit)
158  occurrances.push_back(std::count (dd.begin(), dd.end(), *sit));
159 
160  std::vector<int>::iterator iit = occurrances.begin();
161  sit = dd.begin();
162 
163  std::vector<DataComponentInterpretation::DataComponentInterpretation>
164  data_component_interpretation;
165 
166  for (; iit !=occurrances.end(); ++iit, ++sit)
167  {
168  if (*iit > 1)
169  data_component_interpretation.push_back(DataComponentInterpretation::component_is_part_of_vector);
170  else
171  data_component_interpretation.push_back(DataComponentInterpretation::component_is_scalar);
172  }
173 
174  data_out->add_data_vector (data_vector, dd, DataOut<dim, DoFHandler<dim, spacedim> >::type_dof_data,
175  data_component_interpretation);
176 
177  }
178  deallog << "Added data: " << desc << std::endl;
179  }
180  deallog.pop();
181 }
